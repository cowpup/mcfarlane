Embeddable Iframe Modal

Click the product card below to open the modal:

McFarlane DC Collector Edition Pack

How to Embed

Copy and paste the code below into any HTML page:

<!-- DripShop Modal Embed -->
<div class="dripshop-trigger" onclick="openDripshopModal()" style="
  display: inline-block;
  cursor: pointer;
  background: white;
  border-radius: 12px;
  box-shadow: 0 4px 15px rgba(0,0,0,0.15);
  overflow: hidden;
  transition: transform 0.2s, box-shadow 0.2s;
  max-width: 300px;
  font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
" onmouseover="this.style.transform='translateY(-4px)';this.style.boxShadow='0 8px 25px rgba(0,0,0,0.2)'"
   onmouseout="this.style.transform='translateY(0)';this.style.boxShadow='0 4px 15px rgba(0,0,0,0.15)'">
  <img src="https://cdn.dripshop.live/product/NULUONECgGFN-y63vB9qv.png"
       alt="McFarlane DC Collector Edition Pack" style="width: 100%; height: auto; display: block;">
  <div style="padding: 16px; text-align: center;">
    <span style="font-size: 16px; font-weight: 600; color: #333;">McFarlane DC Collector Edition Pack</span>
  </div>
</div>

<div id="dripshop-modal" style="display: none; position: fixed; top: 0; left: 0; width: 100%; height: 100%; background: rgba(0,0,0,0.8); z-index: 999999; justify-content: center; align-items: center;" onclick="closeDripshopModal(event)">
  <div style="position: relative; width: 95%; max-width: 1380px; height: 85%; background: white; border-radius: 12px; overflow: hidden; box-shadow: 0 20px 60px rgba(0,0,0,0.3);" onclick="event.stopPropagation()">
    <button onclick="closeDripshopModal()" style="position: absolute; top: 10px; right: 10px; width: 40px; height: 40px; border: none; background: rgba(0,0,0,0.7); color: white; font-size: 24px; cursor: pointer; border-radius: 50%; z-index: 10; display: flex; align-items: center; justify-content: center;">&times;</button>
    <iframe id="dripshop-iframe" style="width: 100%; height: 100%; border: none;" allowfullscreen></iframe>
  </div>
</div>

<script>
function openDripshopModal() {
  var modal = document.getElementById('dripshop-modal');
  var iframe = document.getElementById('dripshop-iframe');
  iframe.src = 'https://www.dripshop.live/box-break/cffa4a6f-dd67-40f8-8233-0bcc417069c0';
  modal.style.display = 'flex';
  document.body.style.overflow = 'hidden';
}
function closeDripshopModal(event) {
  if (event && event.target !== document.getElementById('dripshop-modal')) return;
  var modal = document.getElementById('dripshop-modal');
  var iframe = document.getElementById('dripshop-iframe');
  modal.style.display = 'none';
  iframe.src = '';
  document.body.style.overflow = '';
}
document.addEventListener('keydown', function(e) {
  if (e.key === 'Escape') closeDripshopModal();
});
</script>
